\chapter{Appendix: further reading}
\label{ch:further}

\begin{aside}
Books, blog posts, talks that informed this work or expand
beyond it.
\end{aside}

\section{Terminals and protocols}

\begin{itemize}[leftmargin=*]
  \item \emph{The VT100 manual} (DEC, 1978) --- the
    foundation.
  \item Unicode Standard Annex \#11 (East Asian Width),
    \#29 (Grapheme Cluster Breaking).
  \item Kitty terminal protocol docs (kovidgoyal/kitty).
  \item Christian Parpart, Contour terminal blog (synced
    output).
\end{itemize}

\section{C++}

\begin{itemize}[leftmargin=*]
  \item Scott Meyers, \emph{Effective Modern \cpp{}}.
  \item Bjarne Stroustrup, \emph{A Tour of \cpp{}} (3e).
  \item cppreference.com.
\end{itemize}

\section{Reactive programming}

\begin{itemize}[leftmargin=*]
  \item Andre Staltz, \emph{The introduction to Reactive
    Programming you've been missing}.
  \item Solid.js documentation.
  \item Svelte's compiler blog.
\end{itemize}

\section{UI design}

\begin{itemize}[leftmargin=*]
  \item Don Norman, \emph{The Design of Everyday Things}.
  \item Steve Krug, \emph{Don't Make Me Think}.
\end{itemize}
