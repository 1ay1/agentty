\chapter{Appendix: keymap reference}
\label{ch:keyref}

\begin{aside}
The default keymaps for \maya{}'s built-in widgets. Override
via the keymap system (Chapter~\ref{ch:keymaps}).
\end{aside}

\section{Text input}

\begin{itemize}[leftmargin=*]
  \item Left/Right: move cursor one char.
  \item Ctrl-Left/Right: move by word.
  \item Home/End: line ends.
  \item Backspace/Delete: delete char.
  \item Ctrl-W: delete word backward.
  \item Ctrl-U: delete to line start.
  \item Ctrl-K: delete to line end.
  \item Ctrl-A: select all.
  \item Ctrl-C: copy.
  \item Ctrl-X: cut.
  \item Ctrl-V: paste.
  \item Ctrl-Z: undo.
  \item Ctrl-Shift-Z / Ctrl-Y: redo.
\end{itemize}

\section{List / table}

\begin{itemize}[leftmargin=*]
  \item Up/Down: previous/next.
  \item Home/End: first/last.
  \item PgUp/PgDn: viewport jump.
  \item Enter: activate.
  \item Space: toggle select (multi-select mode).
  \item /: focus search.
\end{itemize}

\section{Tree}

\begin{itemize}[leftmargin=*]
  \item Right: expand or first child.
  \item Left: collapse or parent.
  \item Enter: toggle expand.
\end{itemize}

\section{Tabs}

\begin{itemize}[leftmargin=*]
  \item Ctrl-Tab: next.
  \item Ctrl-Shift-Tab: previous.
  \item Ctrl-W: close current.
  \item Ctrl-1..9: jump.
\end{itemize}

\section{Dialog}

\begin{itemize}[leftmargin=*]
  \item Enter: confirm.
  \item Escape: cancel.
  \item Tab: focus next.
\end{itemize}

\section{Global}

\begin{itemize}[leftmargin=*]
  \item F1 or ?: help.
  \item Ctrl-Q: quit.
  \item Ctrl-P: command palette.
  \item Ctrl-L: redraw screen.
\end{itemize}

\section{Recap}

\begin{recap}
\begin{itemize}[leftmargin=*]
  \item Conventions follow Mac/Linux norms.
  \item Override via keymap config.
  \item Document any custom binding in help.
\end{itemize}
\end{recap}
